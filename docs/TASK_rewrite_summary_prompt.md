# Задача: Полностью переписать генерацию саммари
> Файл: backend/app/services/kpi_engine_service.py (или ai_service.py)

## Проблема с текущим подходом
AI сам группирует и считает задачи — делает это неверно:
- Смешивает несвязанные задачи в одну группу
- Придумывает задачи которых нет (галлюцинации)
- Считает неправильно (4 вместо 1)

## Новый подход: Python группирует, AI только пишет текст

### Шаг 1 — Группировка в Python (до промпта)

```python
from collections import Counter

# Словарь ключевых слов для группировки
GROUPS = {
    "разработка": ["разработка", "доработка", "автоматизация", "настройка", 
                   "загрузка", "обновление данных", "добавление", "интеграция"],
    "аналитика":  ["аналитика", "аналитических", "подготовка данных", 
                   "обобщение данных", "подготовка отчёт", "подготовка презентац",
                   "подготовка материал", "подготовка дорожной", "подготовка повестки",
                   "квартального отчет", "обновление данных на дашборд"],
    "совещания":  ["совещание", "встреча", "вкс", "зпмо", "час цифры", 
                   "мероприятия по координации"],
    "защиты":     ["защита", "содоклад"],
    "прочее":     []  # всё остальное
}

def classify_task(subject: str) -> str:
    s = subject.lower()
    for group, keywords in GROUPS.items():
        if group == "прочее":
            continue
        if any(kw in s for kw in keywords):
            return group
    return "прочее"

# Группируем задачи
task_counts = Counter(e['issue']['subject'] for e in work_entries)
groups = {"разработка": [], "аналитика": [], "совещания": [], "защиты": [], "прочее": []}

for task, count in task_counts.items():
    group = classify_task(task)
    groups[group].append((task, count))

# Формируем структурированный текст для промпта
structured_text = ""
group_labels = {
    "разработка": "Разработка и автоматизация",
    "аналитика":  "Аналитика и отчётность", 
    "совещания":  "Совещания и координация",
    "защиты":     "Защиты и презентации",
    "прочее":     "Прочее"
}

for group_key, label in group_labels.items():
    tasks = groups[group_key]
    if not tasks:
        continue
    structured_text += f"\n{label}:\n"
    for task, count in sorted(tasks, key=lambda x: -x[1]):
        if count > 1:
            structured_text += f"  - {task} ({count} раз)\n"
        else:
            structured_text += f"  - {task}\n"
```

### Шаг 2 — Новый промпт (AI только пишет, не считает)

```python
prompt = f"""Ты составляешь деловое саммари для отчёта KPI государственного служащего.

Должность: {role_name}

Выполненные задачи, уже сгруппированные и подсчитанные:

Напиши связное деловое саммари. Правила:

5. Период НЕ упоминать
6. Каждую группу — отдельное предложение или два
7. Если группа пустая — пропусти её
"""
```
