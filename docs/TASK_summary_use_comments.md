# Задача: Переключить саммари на поле comments из трудозатрат
> Файл: backend/app/services/kpi_engine_service.py

## Новая логика источника данных

### Было
Брали `issue.subject` — название задачи → группировали по ключевым словам

### Стало
Берём `comments` — детальное описание что сделано → передаём в AI как есть

### Структура данных для промпта

```python
entries_text = ""
for entry in work_entries:
    subject = entry.get('issue', {}).get('subject', 'Без задачи')
    comment = entry.get('comments', '').strip()
    date = entry.get('spent_on', '')
    hours = entry.get('hours', 0)
    
    if comment and comment.lower() not in ['', subject.lower()]:
        # Есть информативный комментарий
        entries_text += f"[{subject}] {comment}\n"
    else:
        # Комментарий пустой или совпадает с названием — используем только задачу
        entries_text += f"[{subject}]\n"

# Убрать дубли (одинаковые строки)
lines = list(dict.fromkeys(entries_text.strip().split('\n')))
entries_text = '\n'.join(lines)
```

### Новый промпт

```python
prompt = f"""Ты составляешь деловое саммари для отчёта KPI государственного служащего.

```

### Fallback если comments пустые
Если у большинства записей нет комментариев (> 80% пустых) —
вернуть текст-подсказку вместо саммари:

```python
empty_count = sum(1 for e in work_entries 
                  if not e.get('comments', '').strip() 
                  or e.get('comments','').strip() == e.get('issue',{}).get('subject',''))

if empty_count > len(work_entries) * 0.8:
    summary = (
        "Детальные комментарии к трудозатратам не заполнены. "
        "Для корректной AI-оценки рекомендуется указывать в Redmine "
        "описание выполненных работ при списании трудозатрат. "
        f"За период зафиксировано {len(work_entries)} записей трудозатрат."
    )
    # Не вызывать AI, вернуть подсказку
    return summary
```
